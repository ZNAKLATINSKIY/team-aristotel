#include <iostream>
#include <cmath>
using namespace std;

int main() 
{
    int GB;
    int MB = 820;
    int result;
    int multiplication;

    cout << "Сколько у вас гб? на флешке ";
    cin >> GB;
    multiplication = GB * 1024;
    result = multiplication / MB;
    cout << "Вы можете поместить " << floor(result) << " файлов размером 820мб";
    
}
